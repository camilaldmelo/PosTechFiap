Table TBL_CLIENTE {
  ID serial [pk]
  NOME VARCHAR(255)
  CPF VARCHAR(11)
  EMAIL VARCHAR(255)
  DATA DATE
}

Table TBL_CATEGORIA {
  ID serial [pk]
  NOME VARCHAR(255)
}

Table TBL_ACOMPANHAMENTO {
  ID serial [pk]
  NOME VARCHAR(255)
}

Table TBL_PRODUTO {
  ID serial [pk]
  NOME VARCHAR(255)
  PRECO NUMERIC(10, 2)
  DESCRICAO TEXT
  URL_IMAGEM VARCHAR(255)
  ID_CATEGORIA INT [ref: > TBL_CATEGORIA.ID]
}

Table TBL_METODO_PAGAMENTO {
  ID serial [pk]
  NOME VARCHAR(255)
}

Table TBL_PAGAMENTO {
  ID serial [pk]
  ID_PEDIDO INT
  ID_METODO INT [ref: > TBL_METODO_PAGAMENTO.ID]
  VALOR NUMERIC(10, 2)
  DATA_CRIACAO DATE
  DATA_ATUALIZACAO DATE
}

Table TBL_PEDIDO {
  ID serial [pk]
  ID_CLIENTE INT [ref: > TBL_CLIENTE.ID]
  DATA DATE
  ID_ACOMPANHAMENTO INT [ref: > TBL_ACOMPANHAMENTO.ID]
  ID_PAGAMENTO INT [ref: > TBL_PAGAMENTO.ID]
}

Table TBL_PRODUTOS_PEDIDO {
  ID serial [pk]
  ID_PEDIDO INT [ref: > TBL_PEDIDO.ID]
  ID_PRODUTO INT [ref: > TBL_PRODUTO.ID]
  QUANTIDADE INT
}
